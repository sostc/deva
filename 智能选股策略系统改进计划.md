### 3.2 模型优化

#### 3.2.1 集成学习模型

```python
from river import ensemble, linear_model, tree

# 自适应随机森林
class AdaptiveEnsembleModel:
    def __init__(self):
        self.models = {
            'logistic': linear_model.LogisticRegression(),
            'tree': tree.HoeffdingTreeClassifier(),
            'forest': ensemble.AdaptiveRandomForestClassifier(n_models=10),
        }
        self.weights = {name: 1.0 for name in self.models}
    
    def predict_proba(self, features):
        """加权投票预测"""
        predictions = {}
        for name, model in self.models.items():
            pred = model.predict_proba_one(features)
            predictions[name] = pred.get(True, 0.5)
        
        # 加权平均
        weighted_sum = sum(
            pred * self.weights[name] 
            for name, pred in predictions.items()
        )
        total_weight = sum(self.weights.values())
        
        return weighted_sum / total_weight
    
    def update_weights(self, model_name, performance):
        """根据表现更新模型权重"""
        self.weights[model_name] *= (1 + performance)
        # 归一化
        total = sum(self.weights.values())
        self.weights = {k: v/total for k, v in self.weights.items()}
```

#### 3.2.2 模型预热策略

```python
class WarmUpStrategy:
    """模型预热策略，解决初始预测得分偏低问题"""
    
    def __init__(self, warm_up_periods=50):
        self.warm_up_periods = warm_up_periods
        self.current_period = 0
        self.initial_threshold = 0.45  # 预热期低阈值
        self.normal_threshold = 0.55   # 正常期阈值
    
    def get_buy_threshold(self):
        """动态获取买入阈值"""
        if self.current_period < self.warm_up_periods:
            # 预热期：线性增加阈值
            ratio = self.current_period / self.warm_up_periods
            return self.initial_threshold + (self.normal_threshold - self.initial_threshold) * ratio
        return self.normal_threshold
    
    def update(self):
        self.current_period += 1
```

### 3.4 风险管理增强

#### 3.4.1 仓位管理（Kelly公式）

```python
class PositionSizer:
    """仓位管理器"""
    
    def calculate_position_size(self, stock_code, confidence, max_position_pct=0.15):
        """
        使用Kelly公式计算仓位大小
        
        f* = (bp - q) / b
        其中:
        b = 赔率 (平均盈利/平均亏损)
        p = 胜率
        q = 败率 = 1 - p
        """
        # 获取历史交易统计
        win_rate = self.get_win_rate(stock_code)
        avg_win = self.get_average_win(stock_code)
        avg_loss = self.get_average_loss(stock_code)
        
        if avg_loss == 0:
            return max_position_pct
        
        # Kelly公式
        odds = avg_win / abs(avg_loss)
        kelly_fraction = (odds * win_rate - (1 - win_rate)) / odds
        
        # 半Kelly（更保守）
        half_kelly = kelly_fraction * 0.5
        
        # 结合模型置信度
        adjusted_size = half_kelly * confidence
        
        # 限制最大仓位
        return min(adjusted_size, max_position_pct)
```

#### 3.4.2 止盈机制

```python
class TakeProfitManager:
    """止盈管理器"""
    
    def __init__(self):
        self.take_profit_levels = [
            (0.05, 0.30),   # 盈利5%，止盈30%仓位
            (0.10, 0.30),   # 盈利10%，再止盈30%仓位
            (0.15, 0.20),   # 盈利15%，再止盈20%仓位
            (0.20, 0.20),   # 盈利20%，清仓剩余20%
        ]
        self.triggered_levels = {}
    
    def check_take_profit(self, position):
        """检查是否触发止盈"""
        profit_pct = position.profit_pct
        stock_code = position.code
        
        if stock_code not in self.triggered_levels:
            self.triggered_levels[stock_code] = set()
        
        signals = []
        for level, ratio in self.take_profit_levels:
            if profit_pct >= level * 100 and level not in self.triggered_levels[stock_code]:
                signals.append({
                    'action': 'PARTIAL_SELL',
                    'ratio': ratio,
                    'reason': f'盈利达到{level*100}%，止盈{ratio*100}%仓位',
                })
                self.triggered_levels[stock_code].add(level)
        
        return signals
```

#### 3.4.3 止损优化（跟踪止损）

```python
class TrailingStopLoss:
    """跟踪止损"""
    
    def __init__(self, initial_stop=-0.08, trailing_pct=0.05):
        self.initial_stop = initial_stop
        self.trailing_pct = trailing_pct
        self.highest_prices = {}
    
    def update_highest_price(self, stock_code, current_price):
        """更新最高价"""
        if stock_code not in self.highest_prices:
            self.highest_prices[stock_code] = current_price
        else:
            self.highest_prices[stock_code] = max(
                self.highest_prices[stock_code], 
                current_price
            )
    
    def get_stop_loss_price(self, stock_code, buy_price):
        """获取当前止损价"""
        highest = self.highest_prices.get(stock_code, buy_price)
        
        # 初始止损
        initial_stop_price = buy_price * (1 + self.initial_stop)
        
        # 跟踪止损（从最高点回撤）
        trailing_stop_price = highest * (1 - self.trailing_pct)
        
        # 取两者中较高的（更宽松的）
        return max(initial_stop_price, trailing_stop_price)
```

class StrategyPool:
    """策略池"""
    
    def __init__(self):
        self.strategies = {
            'trend_following': TrendFollowingStrategy(),
            'mean_reversion': MeanReversionStrategy(),
            'momentum': MomentumStrategy(),
            'block_rotation': BlockRotationStrategy(),
        }
        self.performance_tracker = PerformanceTracker()
    
    def select_best_strategy(self, market_state):
        """根据市场状态选择最佳策略"""
        performances = {}
        for name, strategy in self.strategies.items():
            perf = self.performance_tracker.get_recent_performance(name, days=20)
            performances[name] = perf
        
        # 根据市场状态选择
        if market_state in ['bull_volatile', 'bull_stable']:
            # 牛市选趋势跟踪
            return self.strategies['trend_following']
        elif market_state == 'sideways':
            # 震荡市选均值回归
            return self.strategies['mean_reversion']
        elif market_state in ['bear_volatile', 'bear_stable']:
            # 熊市选动量策略（做空）
            return self.strategies['momentum']
        else:
            # 默认选表现最好的
            best = max(performances, key=performances.get)
            return self.strategies[best]
